function checkAnswer(answer) { console.log('checkAnswer called with:', answer); var result = document.getElementById('result'); var buttons = document.getElementsByTagName('button'); if (answer == correctAnswers[showCount]) { result.textContent = "Correct answer!"; result.style.color = 'white'; result.style.background = 'green'; for (var i = 0; i < buttons.length; i++) { if (buttons[i].textContent == answer) { buttons[i].style.backgroundColor = 'green'; } } } else { result.textContent = "Incorrect answer."; result.style.color = 'white'; result.style.background = 'red'; for (var i = 0; i < buttons.length; i++) { if (buttons[i].textContent == answer) { buttons[i].style.backgroundColor = 'red'; } } } result.style.display = 'block'; // Show result immediately setTimeout(function() { result.style.display = 'none'; hideButtons(); // Hide buttons after result is displayed }, 5000); // Hide result and buttons after 5 seconds showCount++; console.log('showCount after increment:', showCount); }
